nbaplayerstats2025 <- load_nba_player_box(seasons = 2025, na.rm = TRUE)
nbaplayerstats2024 <- load_nba_player_box(season = 2024)

totalsplayers24 <- nbaplayerstats2024 |> 
  mutate(
    game = as.numeric(as.character.Date(game_date)),
    gamenumb = (is.na(nbaplayerstats2025$game) = 1)
  )

totalsplayers2024 <- totalsplayers24 |>
  group_by(season, athlete_display_name, team_display_name) |>
  summarize(
    totalmin = sum(minutes, na.rm = TRUE),
    totalpts = sum(points, na.rm = TRUE),
    totalreb = sum(rebounds, na.rm = TRUE),
    totalast = sum(assists, na.rm = TRUE),
    totalstl = sum(steals, na.rm = TRUE),
    totalblk = sum(blocks, na.rm = TRUE),
    totalgames = sum(gamenumb, na.rm = TRUE),
    totalfgmade = sum(field_goals_made, na.rm = TRUE),
    totalfgatt = sum(field_goals_attempted, na.rm = TRUE),
    total3ptmade = sum(three_point_field_goals_made, na.rm = TRUE),
    total3ptatt = sum(three_point_field_goals_attempted, na.rm = TRUE),
    totalftmade = sum(free_throws_made, na.rm = TRUE),
    totalftatt = sum(free_throws_attempted, na.rm = TRUE)
  ) |>
  mutate(
    Player = athlete_display_name,
    Team = team_display_name,
    MPG = round((totalmin/totalgames), digits = 1),
    PPG = round((totalpts/totalgames), digits = 1),
    RPG = round((totalreb/totalgames), digits = 1),
    APG = round((totalast/totalgames), digits = 1),
    SPG = round((totalstl/totalgames), digits = 1),
    BPG = round((totalblk/totalgames), digits = 1),
  )

players2024 <- totalsplayers2024 |>
  mutate(
    FGPCT = round((totalfgmade/totalfgatt), digits = 3) * 100,
    THREEPCT = round((total3ptmade/total3ptatt), digits = 3) * 100,
    FTPCT = round((totalftmade/totalftatt), digits = 3) * 100,
  )

players24 <- players2024 |>
  select(season, Team, MPG, PPG, RPG, APG, SPG, BPG, FGPCT, THREEPCT, FTPCT)

players24 <- players24 |>
  mutate(
    AllStats = PPG + RPG + APG + SPG + BPG
  )

totalsplayers <- nbaplayerstats2025 |> 
  mutate(
    game = as.numeric(as.character.Date(game_date)),
    gamenumb = (is.na(nbaplayerstats2025$game) = 1)
  )

totalsplayers2025 <- totalsplayers |>
  group_by(season, athlete_display_name, team_display_name) |>
  summarize(
    totalmin = sum(minutes, na.rm = TRUE),
    totalpts = sum(points, na.rm = TRUE),
    totalreb = sum(rebounds, na.rm = TRUE),
    totalast = sum(assists, na.rm = TRUE),
    totalstl = sum(steals, na.rm = TRUE),
    totalblk = sum(blocks, na.rm = TRUE),
    totalgames = sum(gamenumb, na.rm = TRUE),
    totalfgmade = sum(field_goals_made, na.rm = TRUE),
    totalfgatt = sum(field_goals_attempted, na.rm = TRUE),
    total3ptmade = sum(three_point_field_goals_made, na.rm = TRUE),
    total3ptatt = sum(three_point_field_goals_attempted, na.rm = TRUE),
    totalftmade = sum(free_throws_made, na.rm = TRUE),
    totalftatt = sum(free_throws_attempted, na.rm = TRUE)
  ) |>
  mutate(
    Team = team_display_name,
    MPG = round((totalmin/totalgames), digits = 1),
    PPG = round((totalpts/totalgames), digits = 1),
    RPG = round((totalreb/totalgames), digits = 1),
    APG = round((totalast/totalgames), digits = 1),
    SPG = round((totalstl/totalgames), digits = 1),
    BPG = round((totalblk/totalgames), digits = 1),
  )

eligibleplayers25 <- totalsplayers2025 |>
  filter(MPG > 19.9)

players2025 <- eligibleplayers25 |>
  mutate(
    FGPCT = round((totalfgmade/totalfgatt), digits = 3) * 100,
    THREEPCT = round((total3ptmade/total3ptatt), digits = 3) * 100,
    FTPCT = round((totalftmade/totalftatt), digits = 3) * 100,
  )

players25 <- players2025 |>
